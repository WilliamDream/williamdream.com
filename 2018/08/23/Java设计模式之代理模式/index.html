/**
 * 接口类
 */
public interface IMan {

    public void findGirlFriend();

}
/**
 * 被代理目标对象，实现接口类
 */
public class SingleDog implements IMan{

    @Override
    public void findGirlFriend() {
        System.out.println("me:别看我是个矮穷矬，但是我想找个肤白貌美大长腿，内外兼修气质佳的女朋友!!!");
    }

}

import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;
import java.lang.reflect.Proxy;

/**
 * @author: william
 * @Description: 代理类---珍爱网
 */
public class ZhenAiWang implements InvocationHandler{

    private Object target;


    public Object getInstance(Object target) throws Exception{
        this.target = target;
        Class<?> clazz = target.getClass();
        // 返回指定接口的代理类的实例
        return Proxy.newProxyInstance(clazz.getClassLoader(),clazz.getInterfaces(),this);
    }

    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        // doSomething()
        System.out.println("before:先生您好！我是珍爱网红娘飘飘，请问你想找什么类型的女朋友呢？");
        method.invoke(this.target,args);
        System.out.println("after:屌丝，你怕是还没睡醒吧，滚犊子!!!");
        return null;
    }

}

2、字节码重组实现动态代理

我们已经知道，jdk动态代理是动态的生成一个代理对象的子类来实现的。下面我们通过字节码重组的方式来实现动态代理。

1.代码实现

/**
 * @author: william
 * @Description: 类加载器
 */
public class MyClassLoader extends ClassLoader{

    private File file;

    public MyClassLoader() {
        String classPath = MyClassLoader.class.getResource("").getPath();
        this.file = new File(classPath);
    }

    protected Class<?> findClass(String name) throws ClassNotFoundException {
        String className = MyClassLoader.class.getPackage().getName() +"."+name;
        if(file != null) {
            File classFile = new File(file, name.replace("\\./", "/")+".class");
            if(classFile.exists()) {
                FileInputStream in = null;
                ByteArrayOutputStream out = null;

                try {
                    in = new FileInputStream(classFile);
                    out = new ByteArrayOutputStream();
                    byte[] buf = new byte[1024];
                    int len;
                    while((len = in.read(buf)) != -1) {
                        out.write(buf,0,len);
                    }
                    return defineClass(className,out.toByteArray(),0,out.size());
                } catch (Exception e) {
                    e.printStackTrace();
                } finally {
                    if(null != in){
                        try {
                            in.close();
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }

                    if(out != null){
                        try {
                            out.close();
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }
                }

            }
        }
        return null;
    }


}

import java.io.File;
import java.io.FileWriter;
import java.lang.reflect.Constructor;
import java.lang.reflect.Method;

import javax.tools.JavaCompiler;
import javax.tools.StandardJavaFileManager;
import javax.tools.ToolProvider;

/**
 * @author: chaiz
 * @Description: 自己的代理类，用于生成java对象并编译加载到jvm中
 */
public class MyProxy {

    public static final String ln = "\r\n";

    @SuppressWarnings({ "rawtypes", "unchecked" })
    public static Object newProxyInstance(MyClassLoader classLoader,Class<?> [] interfaces,MyInvocationHandler h) {
        try {
            // 1.动态生成源代码java文件
            String javaStr = getJavaStr(interfaces);

            // 2.将java文件输出到磁盘
            String filePath = MyProxy.class.getResource("").getPath();
            File javaFile = new File(filePath + "$Proxy0.java");
            FileWriter fw = new FileWriter(javaFile);
            fw.write(javaStr);
            fw.flush();
            fw.close();

            // 3.将java文件编译为class文件
            JavaCompiler jc = ToolProvider.getSystemJavaCompiler();
            StandardJavaFileManager manager = jc.getStandardFileManager(null, null, null);
            Iterable iterable = manager.getJavaFileObjects(javaFile);
            JavaCompiler.CompilationTask task = jc.getTask(null, manager, null, null, null, iterable);
            task.call();
            manager.close();

            // 4.将class文件加载到JVM中
            Class proxyClass =  classLoader.findClass("$Proxy0");
            Constructor c = proxyClass.getConstructor(MyInvocationHandler.class);
            //删除生成的java类
            javaFile.delete();

            // 5.返回字节码重组后生成的代理对象
            return c.newInstance(h);
        } catch (Exception e) {
            e.printStackTrace();
        }
        return null;
    }

    /**
     * @Title: getJavaStr
     * @Description: 自己拼接一个java文件，原本是有代理生成的对象
     * @param interfaces
     * @return
     */
    public static String getJavaStr(Class<?>[] interfaces) {
        StringBuffer sb = new StringBuffer();
        sb.append("package imcomp.classproxy; " + ln);
        sb.append("import imcomp.classproxy.IMan; " + ln);
        sb.append("import java.lang.reflect.Method; " + ln);
        sb.append("public class $Proxy0 implements " + interfaces[0].getName() + "{" + ln );
            sb.append("MyInvocationHandler h; " + ln );

            sb.append("public $Proxy0(MyInvocationHandler h) { ");
                sb.append("this.h = h;" + ln);
            sb.append("} " + ln);
            for (Method m : interfaces[0].getMethods()){
                sb.append("public " + m.getReturnType().getName() + " " + m.getName() + "() {" + ln);
                    sb.append("try{" + ln);
                        sb.append("Method m = " + interfaces[0].getName() + ".class.getMethod(\"" + m.getName() + "\",new Class[]{});" + ln);
                        sb.append("this.h.invoke(this,m,null);" + ln);
                    sb.append("}catch(Throwable e){" + ln);
                        sb.append("e.printStackTrace();" + ln);
                    sb.append("} " + ln);
                sb.append("} " + ln);
            }
        sb.append("} " + ln);

        return sb.toString();
    }

}

import java.lang.reflect.Method;

/**
 * @author: william
 * @Description: 编写模拟回调的动态代理类
 */
public interface MyInvocationHandler {

    public Object invoke(Object proxy, Method method, Object[] args)
            throws Throwable;
}

/**
 * @author: chaiz
 * @Description: 代理类---珍爱网
 */
public class ZhenAiWang implements MyInvocationHandler{

    private IMan target;


    public Object getInstance(IMan target) throws Exception{
        this.target = target;
        Class<?> clazz = target.getClass();
        // 返回指定接口的代理类的实例
        return MyProxy.newProxyInstance(new MyClassLoader(),clazz.getInterfaces(),this);
    }

    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {

        System.out.println("before:先生您好！我是珍爱网红娘飘飘，请问你想找什么类型的女朋友呢？");
        method.invoke(target, args);
        System.out.println("after:屌丝，你怕是还没睡醒吧，滚犊子!!!");
        return null;
    }

}
