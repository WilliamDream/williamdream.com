Elasticsearch聚合查询之一简单聚合

1. 聚合查询分类

  • 过滤聚合

2. 简单聚合

2.1 统计平均值
# 统计所有书籍的平均价格
GET /library/books/_search
{
    "size":0,
    "aggregations":{
        "avg_price":{
            "avg":{
                "field":"price"
            }
        }
    }
}

# 根据书籍不同类别分组统计每个版本的平均价格
# select type,avg(pirce) from books group by type
GET /library/books/_search
{
    "size":0,
    "aggs":{
        "genders":{
            "terms":{
                "field":"type"
            },
            "aggs":{
                "avg_price":{
                    "avg":{
                        "field":"pirce"
                    }
                }
            }
        }
    }
}
2.2 统计最大值与最小值

GET /library/books/_search
{
  "size":0,
    "aggregations":{
        "min_price":{
            "min":{
                "field":"price"
            }
        },
        "max_price":{
            "max":{
                "field":"price"
            }
        }
    }
}
